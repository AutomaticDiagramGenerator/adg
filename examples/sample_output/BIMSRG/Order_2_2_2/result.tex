\section{Permutators definitions}

\begin{align*}
P(k_{1}/k_{2}) &= 1 - P_{k_{1} k_{2}} \\
P(k_{1}/k_{2}k_{3}) &= 1 - P_{k_{1} k_{2}} - P_{k_{1} k_{3}} \\
P(k_{1}/k_{2}k_{3}k_{4}) &= 1 - P_{k_{1} k_{2}} - P_{k_{1} k_{3}} - P_{k_{1} k_{4}} \\
P(k_{1}k_{2}/k_{3}k_{4}) &= 1 - P_{k_{1} k_{3}} - P_{k_{1} k_{4}} - P_{k_{2} k_{3}} - P_{k_{2} k_{4}} + P_{k_{1} k_{3}} P_{k_{2} k_{4}} + P_{k_{2} k_{3}} P_{k_{1} k_{4}} \\
\end{align*}
